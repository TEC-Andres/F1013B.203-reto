\inicializarSubseccion{Funciones de MATLAB utilizadas}

\begin{center}
\begin{tabular}{|l|l|}
\hline
\textbf{Función} & \textbf{Uso} \\ \hline
\verb|linspace()| & Genera los vectores de puntos en $x$ e $y$ \\ \hline 
\verb|meshgrid()| & Crea la malla 2D del espacio de simulación \\ \hline
\verb|sqrt()| & Calcula distancias y magnitud del campo \\ \hline
\verb|quiver()| & Grafica los vectores del campo eléctrico \\ \hline
\verb|plot()| & Marca la posición de las cargas en la figura \\ \hline
\verb|text()| & Agrega las etiquetas $+q$ y $-q$ \\ \hline
\verb|sprintf()| & Muestra los parámetros en el título de la figura \\ \hline
\verb|title()|, \verb|xlabel()|, \verb|ylabel()| & Da formato y nombres a los ejes \\ \hline
\verb|axis()|, \verb|grid()| & Ajusta los límites y activa la cuadrícula \\ \hline
\end{tabular}
